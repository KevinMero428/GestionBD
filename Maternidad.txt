/* Table: CITAS                                                 */
/*==============================================================*/
create table CITAS (
   IDCITAS               int                 not null,
   ID_PACIENTE           int	         	   null,
   ID_DOCTOR             int        		   null,
   CITA_FECHA	      	date                null,
   CITA_HORA            time                null,
   
   constraint PK_CITAS primary key (IDCITAS)
);

/*==============================================================*/
/* Table: CONSULTAS                                             */
/*==============================================================*/
create table CONSULTAS (
   IDCONSULTAS           int                 not null,
   ID_PACIENTE           int	         	   null,
   ID_DOCTOR             int		            null,
   CONSULTAS_MOTIVO		 varchar(50)		      null,
   constraint PK_CONSULTAS primary key (IDCONSULTAS)
);

/*==============================================================*/
/* Table: DOCTOR                                              */
/*==============================================================*/
create table DOCTOR (
   IDDOCTOR             int                  not null,
   DOCTOR_NOMBRE        varchar(50)          null,
   DOCTOR_APELLIDOS     varchar(50)          null,
   DOCTOR_CELL       	varchar(10)          null,
   DOCTOR_CORREO        varchar(50)          null,
   DOCTOR_ESPECIALIDAD 	varchar(50)          null,
   constraint PK_DOCTOR primary key (IDDOCTOR)
);

);
